def checkPermutation(str1, str2):
    str1 = str(str1)
    str2 = str(str2)
    if len(str1) != len(str2):
        return False
    for i in str1:
        if i not in str2:
            return False
    for i in str2:
        if i not in str1:
            return False
    return True
